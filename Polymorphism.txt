import java.util.Scanner;

class Publication {
    String title;
    int price, copies;

    void sellCopies() {
        System.out.println("Total sale: " + (copies * price));
    }
}

class Book extends Publication {
    String author;

    void sellCopies() {
        price = 300;
        System.out.println("Book total sale: " + (copies * price));
    }

    void orderCopies() {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter the number of copies of the book: ");
        copies = sc.nextInt(); 
    }
    void display(){
        Scanner sc = new Scanner(System.in);
        System.out.println("enter the name of book:");
        String title = sc.nextLine();
    }
}

class Magazine extends Publication {
    void sellCopies() {
        price = 500;

        System.out.println("Magazine total sale: " + (copies * price));
    }

    void orderCopies() {
        Scanner sc =new Scanner(System.in);
        System.out.print("Enter the number of copies of the magazine: ");
        copies = sc.nextInt(); 
    }

    void currentIssue() {
        System.out.println("Displaying the current issue of the magazine.");
    }

    void receiveIssue() {
        System.out.println("Receiving the issue of the magazine.");
    }
    void display(){
        Scanner sc = new Scanner(System.in);
        System.out.println("enter the name of magazine:");
        String title = sc.nextLine();
    }
}

public class Main {
    public static void main(String[] args) {
        Book b1 = new Book();
        b1.display();
        
        
        b1.orderCopies();
        b1.sellCopies();

        Magazine m1 = new Magazine();
        m1.display();
    
        
        m1.orderCopies();
        m1.sellCopies();
    }
}
